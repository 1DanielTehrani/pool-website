Usage

   Choose a miner listed abvove. Download a Zipfile of it and edit the configuration file as shown. You can adjust the gpu usage by changing the amount of threads and blocks to use. 

//config.txt
"gpu_threads_conf" : [
  { "index" : 0,
    "threads" : 64,
    "blocks"  : 8,  //optional
    "bfactor" : 8, //optional
    "bsleep" :  100, //optional
    "affine_to_cpu" : false,
  },
    ],

"use_tls" : false,
"tls_secure_algo" : true,
"tls_fingerprint" : "",


"pool_address" : "127.0.0.1:8888",
"wallet_address" : "your wallet address", //ex.43VeqbfqScY5qk6x14VvUNZLiMkpxpdKU7YAyz2ukuoPAfNL74aUpa5TKDhPV37VzdeJWr6wjnP4bbusYp5gQYE9Ca6ryy1
"pool_password" : "x",


"call_timeout" : 10,
"retry_time" : 10,
"giveup_limit" : 0,


"verbose_level" : 3,


"h_print_time" : 60,


"output_file" : "",


"httpd_port" : 0,

"prefer_ipv4" : true,
